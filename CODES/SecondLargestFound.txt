public class Main {
    public static void main(String args[]) {
        int arr[] = {10, 20, 4, 40, 30};

        // Initialize the highest and second highest elements
        int highest = Integer.MIN_VALUE;
        int secondHighest = Integer.MIN_VALUE;

        // Iterate through the array
        for (int i = 0; i < arr.length; i++) {
            // If the current element is greater than the highest element, update the highest and second highest elements
            if (arr[i] > highest) {
                secondHighest = highest;
                highest = arr[i];
            }
            // If the current element is greater than the second highest element but less than the highest element, update the second highest element
            else if (arr[i] > secondHighest && arr[i] < highest) {
                secondHighest = arr[i];
            }
        }

        System.out.println("The second highest element is: " + secondHighest);
    }
}
